99/* Size definition for CPU sets. */
100# define __CPU_SETSIZE 1024
101# define __NCPUBITS (8 * sizeof (__cpu_mask))
102
103/* Type for array elements in 'cpu_set'. */
104typedef unsigned long int __cpu_mask;
105
106/* Basic access functions. */
107# define __CPUELT(cpu) ((cpu) / __NCPUBITS)
108# define __CPUMASK(cpu) ((__cpu_mask) 1 << ((cpu) % __NCPUBITS))
109
110/* Data structure to describe CPU mask. */
111typedef struct
112{
113 __cpu_mask __bits[__CPU_SETSIZE / __NCPUBITS];
114} cpu_set_t;
115
116/* Access functions for CPU masks. */
117# define __CPU_ZERO(cpusetp) \
118 do { \
119 unsigned int __i; \
120 cpu_set_t *__arr = (cpusetp); \
121 for (__i = 0; __i < sizeof (cpu_set_t) / sizeof (__cpu_mask); ++__i) \
122 __arr->__bits[__i] = 0; \
123 } while (0)
124# define __CPU_SET(cpu, cpusetp) \
125 ((cpusetp)->__bits[__CPUELT (cpu)] |= __CPUMASK (cpu))
126# define __CPU_CLR(cpu, cpusetp) \
127 ((cpusetp)->__bits[__CPUELT (cpu)] &= ~__CPUMASK (cpu))
128# define __CPU_ISSET(cpu, cpusetp) \
129 (((cpusetp)->__bits[__CPUELT (cpu)] & __CPUMASK (cpu)) != 0)
130#endif
